- lastName          -> String
- otherName         -> String
- type              -> enum["admin", "teacher", "student"]
- profile            -{
                      id:   -> String
                      url:  -> String
                      }
- createAt          -> Date
- updatedAt         -> Date
- email              -> String
- phone             -> String
- plan              -> enum["Free", "School", "Premium"]
